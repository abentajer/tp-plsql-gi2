CREATE PROCEDURE BDE_gagnants IS

	v_nom Electeurs.nom%type ;
	v_pnom Electeurs.pnom%type ;
	v_nvoix Electeurs.nvoix%type ;

BEGIN

	SELECT nom ,pnom, nvoix
	FROM Electeurs
	ORDER BY nvoix

	FETCH sql into v_nom ,v_pnom ,v_nvoix ;
	dbms_output.put_line(1 || '-nom: '|| v_nom  || ' prenom: ' || v_pnom || 'nobre de vote : ' || v_nvoix);

END;
